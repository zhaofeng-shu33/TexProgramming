%Program: output Fibonacci numbers within the first 20 to the stdout
\newcount\a
\a=1
\newcount\b
\b=1
\newcount\c
\newcount\i
\i=1
\newlinechar = 10%define the new line character to \r or ^^J
\message{^^J Output: ^^J}%\message command write to stdout
\loop
  \message{\i : \the\a ^^J}
\ifnum\i < 21
\advance\i by 1
\advance\b by \c
\c=\the\a
\a=\the\b
\repeat
\end%end of the program
